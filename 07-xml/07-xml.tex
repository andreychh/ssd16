  \vspace*{12pt}
  \ff{/library/book[@id='42']}\par
  \ff{//book[@id='42']}\par
  \ff{//book[first()]}\par
  \ff{//book[author='David West']}\par
  \ff{//book[author[text()='David West']]}
}
