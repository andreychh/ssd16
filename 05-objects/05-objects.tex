\section[Object]{What is an Object?}

\innoPlush{
  \subsection{C++}
  \innoCite{stroustrup}{An object is some memory that holds a value of some type}{\emph{Programming Principles and Practice Using C++} by \nospell{Bjarne Stroustrup}}
}

\innoPlush{
  \subsection{Wiki}
  \innoCite{wikipedia}{Objects may contain data, in the form of fields, often known as attributes; and code, in the form of procedures, often known as methods}{Wikipedia}
}

\innoPlush{
  \subsection{Smalltalk}
  \innoCite{smalltalk}{An object consists of some private memory and a set of operations}{\emph{Smalltalk-80: The Language and Its Implementation} by \nospell{Adele Goldberg} et al., p.~6}
}

\innoPlush{
  \subsection{Java}
  \innoCite{java-nutshell}{A class is a collection of data fields that hold values and methods that operate on those values}{\emph{Java in a Nutshell} by \nospell{Ben Evans}}
}

\innoPlush{
  \subsection{\nospell{Eckel}}
  \innoCite{eckel}{Each object looks quite a bit like a little computer --- it has a state, and it has operations that you can ask it to perform}{\emph{Thinking in Java} by \nospell{Bruce Eckel}, p.~16}
}

\innoPlush{
  \subsection{West}
  \innoCite{object-thinking}{An object is the equivalent of the quanta from which the universe is constructed}{\emph{Object Thinking} by \nospell{David West}, p.~66}
}
